18#ifndef MIR_VERSION_NUMBER_H_
19#define MIR_VERSION_NUMBER_H_
20
21/**
22 * MIR_VERSION_NUMBER
23 * \param major [in] The major version (eg: 3 for version 3.2.33)
24 * \param minor [in] The minor version (eg: 2 for version 3.2.33)
25 * \param micro [in] The micro version (eg: 33 for version 3.2.33)
26 *
27 * Returns the combined version information as a single 32-bit value for
28 * logical comparisons. For example:
29 * \#if MIR_CLIENT_VERSION >= MIR_VERSION_NUMBER(2,3,4)
30 *
31 * This can be useful to conditionally build code depending on new features or
32 * specific bugfixes in the Mir client library.
33 */
34#define MIR_VERSION_NUMBER(major,minor,micro)
35 (((major) << 22) + ((minor) << 12) + (micro))
36
37#endif /* MIR_VERSION_NUMBER_H_ */
